#include<stdio.h>
int main(){
    int bal,N,i=0,p,success=0;
    scanf("%d",&bal);
    scanf("%d",&N);

    while(i<N){
        scanf("%d",&p);
        if(bal >= p){
            bal -= p;
            success++;
        } else break;
        i++;
    }

    printf("Successful Purchases: %d\n",success);
    printf("Final Balance: %d",bal);
    return 0;
}
