#include<stdio.h>
int main(){
    int cap,N,i=0,load,safe=0,fail=0;
    scanf("%d",&cap);
    scanf("%d",&N);

    while(i<N){
        scanf("%d",&load);
        if(load > cap) fail++;
        else safe++;
        i++;
    }

    printf("Safe Hours: %d\n",safe);
    printf("Failure Count: %d",fail);
    return 0;
}
