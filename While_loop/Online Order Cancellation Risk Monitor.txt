#include<stdio.h>
int main(){
    int num;
    scanf("%d",&num);
    int Success=0;
    int Cancelled=0;
    int i=1;
    while(i<=num){
        int inp;
        scanf("%d",&inp);
        if(inp==0){
            Cancelled=Cancelled+1;
        }
        else{
            Success=Success+1;
        }
        i++;
    }
    printf("Success : %d\n",Success);
    printf("Cancelled : %d\n",Cancelled);
    if(Success>=Cancelled){
        printf("Status : Safe");
    }
    else{
        printf("Status : Risk");
    }
    return 0;
}
