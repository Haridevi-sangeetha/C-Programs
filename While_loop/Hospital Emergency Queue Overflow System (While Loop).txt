#include<stdio.h>
int main(){
    int cap,N,i=0,arrive,treated=0,reject=0;
    scanf("%d",&cap);
    scanf("%d",&N);

    while(i<N){
        scanf("%d",&arrive);
        if(treated + arrive <= cap){
            treated += arrive;
        } else {
            reject += (treated + arrive) - cap;
            treated = cap;
        }
        i++;
    }

    printf("Treated Patients: %d\n",treated);
    printf("Rejected Patients: %d",reject);
    return 0;
}
