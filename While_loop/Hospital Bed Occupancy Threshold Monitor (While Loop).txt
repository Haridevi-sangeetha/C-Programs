#include<stdio.h>
int main(){
    int cap,N,i=0,chg,occ=0,critical=0;
    scanf("%d",&cap);
    scanf("%d",&N);

    while(i<N){
        scanf("%d",&chg);
        occ += chg;
        if(occ > (cap*90)/100)
            critical++;
        i++;
    }

    printf("Final Occupied Beds: %d\n",occ);
    printf("Critical Hours: %d",critical);
    return 0;
}
