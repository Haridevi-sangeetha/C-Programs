#include<stdio.h>
int main(){
    int N,i=0,p,max=0,surge=0;
    scanf("%d",&N);

    while(i<N){
        scanf("%d",&p);
        if(p > max)
            max = p;
        if(p > 5)
            surge++;
        i++;
    }

    printf("Max Usage: %d\n",max);
    printf("Surge Hours: %d",surge);
    return 0;
}
