#include<stdio.h>
int main(){
    int N,i=0,x,cur=0,max=0;
    scanf("%d",&N);

    while(i<N){
        scanf("%d",&x);
        if(x==0){
            cur++;
            if(cur>max) max=cur;
        }
        else{
            cur=0;
        }
        i++;
    }

    printf("Longest Default Streak: %d",max);
    return 0;
}
