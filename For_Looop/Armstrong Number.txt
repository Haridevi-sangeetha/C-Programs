
#include<stdio.h>

int main(){
    int num;
    scanf("%d",&num);
    int sum =0;
    int temp = num;
    for(;num!=0;){
        int id=num%10;
        int cube = id*id*id;
        sum = sum+cube;
        num = num/10;
    }
    if(sum==temp){
        printf("Yes");
    }
    else{
        printf("No");
    }
    return 0;
}
