#include<stdio.h>

int main(){
    int num1,num2;
    scanf("%d %d",&num1,&num2);
    int min;
    if(num1>num2){
        min=num2;
    }
    else{
        min=num1;
    }
    int gcd;
    for(int i=1;i<=min;i++){
        if(num1%i==0  && num2%i==0){
            gcd =i;
        }
    }
    int lcm = (num1*num2)/gcd;
    printf("%d",lcm);
    return 0;
}
