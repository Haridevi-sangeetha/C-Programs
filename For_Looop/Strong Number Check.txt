#include<stdio.h>

int main(){
    int num;
    scanf("%d",&num);
    int sum=0;
    // int fact=1;
    int temp=num;
    for(;num!=0;){
        int id=num%10;
        int fact=1;
        for(int i=1;i<=id;i++){
            fact=fact*i;
        }
        sum=sum+fact;
        num=num/10;
    }
    if(sum==temp){
        printf("YES");
    }
    else{
        printf("NO");
    }
    return 0;
}
