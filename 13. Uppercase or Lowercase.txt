#include<stdio.h>

int main(){
    char ch;
    scanf("%c",&ch);
    if(ch>='A' && ch<='Z'){
        printf("%c is UpperCase.",ch);
    }
    else if(ch>='a' && ch<='z'){
        printf("%c is Lowercase.",ch);
    }else{
        printf("Please Enter a Valid character");
    }
    return 0;
}
