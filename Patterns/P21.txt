
#include<stdio.h>

int main(){
    int num;
    scanf("%d",&num);
    for(int i=0;i<num;i++){
        for(int j=0;j<=i;j++){
            if(i==j ||i==num-1 || j==0){
                printf("* ");
            }
            else{
                printf("  ");
            }
        }
        printf("\n");
    }
}
