#include<stdio.h>

int main(){
    int num;
    scanf("%d",&num);
    int m = 1;
    int k = 1;
    for(int i=0;i<num;i++){
        for(int j=0;j<=i;j++){
            if(i==j){
                printf ("%d ",m++);
            }
            else if(i==num-1){
                printf("%d ",k++);
            }
            else if (i!=0 && j==0){
                printf("1 ");
            }
            else{
                printf("  ");
            }
        }
        printf("\n");
    }
    return 0;
}
