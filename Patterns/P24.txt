#include<stdio.h>

int main(){
    int num;
    scanf("%d",&num);
    int h = 65;
    int k = 65;
    for(int i=0;i<num;i++){
        for(int j=0;j<=i;j++){
            if(i==j){
                printf("%c ",h++);
            }
            else if(i==num-1){
                printf("%c ",k++);
            }
            else if(i!=0 && j==0 && i!=num-1){
                printf("A ");
            }
            else{
                printf("  ");
            }
        }
        printf("\n");
    }
    return 0;
}
