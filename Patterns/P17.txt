#include<stdio.h>

int main(){
    int num;
    scanf("%d",&num);
    for(int i=1;i<=num;i++){
        int ch='A';
        for(int j=num;j>i;j--){
            printf("  ");
        }
        for(int j=1;j<i;j++){
            printf("%c ",ch++);
        }
        for(int j=i-2;j>0;j--){
            printf("%c ",j+64);
        }
        printf("\n");
    }
    return 0;
}
