#include <stdio.h>

int main() {
    char str[101];
    scanf("%s", str);
    char *start = str;
    char *end = str;
    while (*end != '\0') {
        end++;
    }
    end--;
    while (start < end) {
        if (*start != *end) {
            printf("NO");
            return 0;
        }
        start++;
        end--;
    }

    printf("YES");
    return 0;
}
