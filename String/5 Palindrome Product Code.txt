
#include<stdio.h>
#include<string.h>

int main(){
    char str[100];
    fgets(str,sizeof(str),stdin);
    int len=strlen(str);
    if(str[len-1]=='\n'){
        str[len-1]=='\0';
        len--;
    }
    int palindrome=1;
    for(int i=0;i<len/2;i++){
        if(str[i]!=str[len-1-i]){
            palindrome=0;
            break;
        }
    }
    if(palindrome){
        printf("Palindrome");
    }
    else{
        printf("Not a Palindrome");
    }
    return 0;
}
