
#include<stdio.h>

int main(){
    int row,col;
    scanf("%d %d",&row,&col);
    int arr[row][col];
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    int count ;
    int max_count = 0;
    for(int i=0;i<row;i++){
        // count = 0;
        for(int j=0;j<col;j++){
            int first=arr[i][j];
            count = 0;
            // printf("%d ",first);
            for(int k=2;k<=first;k++){
                // printf("Value of k : %d\n",k);
                if(first%k == 0){
                    count +=1;
                    // printf("k : %d , count : %d\n",k,count);
                }
            }
            if(count == 1){
                max_count+=1;
            }
        }
    }
    printf("Total prime number count : %d",max_count);
    return 0;
}
