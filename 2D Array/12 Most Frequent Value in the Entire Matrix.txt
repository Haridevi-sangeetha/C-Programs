#include<stdio.h>

int main(){
    int row,col;
    scanf("%d %d",&row,&col);
    int arr[row][col];
   
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    int count ;
    int max_count=0;
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            count = 0;
            for(int k=0;k<row;k++){
                for(int l=0;l<col;l++){
                    if(arr[i][j]==arr[k][l]){
                        count+=1;
                    }
                }
            }
            if(count>max_count){
                max_count = arr[i][j];
            }
        }
    }
    printf("%d",max_count);
    return 0;
}
