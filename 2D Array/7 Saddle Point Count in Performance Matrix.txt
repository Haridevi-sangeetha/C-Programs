
#include<stdio.h>
#include<limits.h>

int main(){
    int row,col;
    scanf("%d %d",&row,&col);
    int arr[row][col];
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    int small = INT_MAX;
    for(int i=0;i<row;i++){
        int min = INT_MAX;
        for(int j=0;j<col;j++){
            if(arr[i][j]<min){
                min = arr[i][j];
            }
        }
        if(min<small){
            small=min;
        }
    }
    printf("Total Min : %d",small);
    return 0;
}
