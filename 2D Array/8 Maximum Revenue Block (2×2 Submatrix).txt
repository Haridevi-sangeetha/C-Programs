
#include<stdio.h>
#include<limits.h>

int main(){
    int row,col;
    scanf("%d %d",&row,&col);
    int arr[row][col];
    for(int i=0;i<(row);i++){
        for(int j=0;j<(col);j++){
            scanf("%d",&arr[i][j]);
        }
    }
    int sum ;
    int max_sum= INT_MIN;
    for(int i=0;i<(row-1);i++){
        sum = 0;
        for(int j=0;j<(col-1);j++){
            sum = arr[i][j]+arr[i][j+1]+arr[i+1][j]+arr[i+1][j+1];
            // printf("Sum : %d\n",sum);
            if(sum>max_sum){
                max_sum=sum;
            }
        }
    }
    printf("%d",max_sum);
    return 0;
}
