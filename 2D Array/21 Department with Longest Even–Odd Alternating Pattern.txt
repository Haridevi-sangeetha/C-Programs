#include<stdio.h>

int main(){
    int row,col;
    scanf("%d %d",&row,&col);
    int arr[row][col];
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    int odd_count=0;
    int even_count=0;
    int max_odd_count=0;
    int max_even_count=0;
    int count=0;
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            int first=arr[i][j];
            if(first%2==0){
                even_count++;
            }
            else{
                odd_count++;
            }
        }
        if(even_count>max_even_count && odd_count>max_odd_count){
            max_even_count=even_count;
            max_odd_count=odd_count;
            count=i;
        }
    }
    printf("Count : %d",count);
    return 0;
}
