#include<stdio.h>

int main(){
    int row,col;
    scanf("%d %d",&row,&col);
    int arr[row][col];
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            scanf("%d",&arr[i][j]);
        }
    }

    int max = 0;
    int count;
    for(int i=0;i<row;i++){
        count = 0;
        for(int j=0;j<col-1;j++){
            int first=arr[i][j];
            if(first<arr[i][j+1]){
                first=arr[i][j+1];
                count+=1;
                printf("first : %d ,arr[i][j+1] : %d , Row_count : %d\n",first,arr[i][j+1],count);
            }
        }
        if(count==(col-1)){
            max+=1;
        }
        // printf("%d",max);
    }
    printf("%d",max);
    return 0;
}
