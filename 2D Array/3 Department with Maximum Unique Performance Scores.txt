
#include<stdio.h>

int main(){
    int row,col;
    scanf("%d %d",&row,&col);
    int arr[row][col];
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    int count =0;
    int max_count =0 ;
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            if(arr[i][j]==arr[i][j+1]){
                count +=1;
                // printf("%d %d\n",arr[i][j],count);
                if(count>max_count){
                    max_count=arr[i][j];
                    // printf("max_count : %d\n",arr[i][j]);
                }
            }
        }
        // printf("\n");
    }
    printf("%d",max_count);
    return 0;
}
