
#include<stdio.h>

int main(){
    int row,col;
    scanf("%d %d",&row,&col);
    int arr[row][col];
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    int sum = 0;
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            if(i==0 || i==(row-1) || j==0 || j==(col-1)){
                sum = sum+arr[i][j];
            }
        }
    }
    printf("Sum : %d",sum);
    return 0;
}
