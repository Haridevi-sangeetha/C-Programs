
#include<stdio.h>

int main(){
    int row,col;
    scanf("%d %d",&row,&col);
    int arr[row][col];
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    int product;
    int max_product=0;
    int count=0;
    for(int i=0;i<row;i++){
        product=1;
        for(int j=0;j<col;j++){
            product=product*arr[i][j];
        }
        // printf("Row_Wise Product : %d",product);
        if(product>max_product){
            max_product=i;
            count+=1;
        }
    }
    printf("Max_Product : %d",max_product);
    return 0;
}
