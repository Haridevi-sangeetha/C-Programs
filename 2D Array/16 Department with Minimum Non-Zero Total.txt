#include<stdio.h>
#include<limits.h>

int main(){
    int row,col;
    scanf("%d %d",&row,&col);
    int arr[row][col];
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    int sum = 0;
    int min_sum = INT_MAX;
    int min=0;
    for(int i=0;i<row;i++){
        sum = 0;
        for(int j=0;j<col;j++){
            sum = sum+arr[i][j];
        }
        if(sum!=0  && sum<min_sum){
            min_sum = sum;
            min=i;
        }
        printf("Sum : %d ",sum);
    }
    printf("\nMin_sum : %d",min);
    return 0;
}
