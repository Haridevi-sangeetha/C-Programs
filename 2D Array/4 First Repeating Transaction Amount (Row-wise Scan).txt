#include<stdio.h>

int main(){
    int row,col;
    scanf("%d %d",&row,&col);
    int arr[row][col];
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            int first = arr[i][j];
            int count = 0;
            int cols = j+1;
            for(int k=i;k<row;k++){
                for(int m=cols;m<col;m++){
                    if(first==arr[k][m]){
                        count+=1;
                        printf("%d",first);
                    }
                }
                cols=0;
            }
        }
    }
}
