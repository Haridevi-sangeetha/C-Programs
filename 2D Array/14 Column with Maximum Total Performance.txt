#include<stdio.h>

int main(){
    int row,col;
    scanf("%d %d",&row,&col);
    int arr[row][col];
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    int sum;
    int max_sum = 0;
    for(int j=0;j<col;j++){
        sum =0;
        for(int i=0;i<row;i++){
            sum = sum+arr[i][j];
        }
        printf("%d ",sum);
        if(sum>max_sum){
            max_sum = j;
        }
    }
    printf("\nSum : %d",max_sum);
    return 0;
}
