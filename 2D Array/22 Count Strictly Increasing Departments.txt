#include<stdio.h>

int main(){
    int row,col;
    scanf("%d %d",&row,&col);
    int arr[row][col];
    for(int i=0;i<row;i++){
        for(int j=0;j<col;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    int count;
    int max_count =0;
    for(int i=0;i<row;i++){
        count = 0;
        for(int j=0;j<(col-1);j++){
            if(arr[i][j]<arr[i][j+1]){
                count++;
            }
            // printf("arr[i][j] : %d , arr[i][j+1] : %d , count : %d\n",arr[i][j],arr[i][j+1],count);
        }
        if(count == (col-1)){
            max_count+=1;
        }
    }
    printf("%d",max_count);
    return 0;
}
