#include<stdio.h>

int main(){
    int con_type;
    scanf("%d",&con_type);
    int units;
    scanf("%d",&units);
    switch(con_type){
        case 1:
            if(units<=100){
                int cal = units*3;
                printf("%d Bil : ",cal);
            }
            else if(units>=100 && units<=200){
                int cal = (100*3)+((units-100)*5);
                printf("%d Bill : ",cal-80);
            }
            else if(units>=200){
                int cal =(100*3)+(100*5)+((units-200)*7);
                printf("%d Bill : ",(cal-80));
            }
            break;
        case 2:
            if(units<=100){
                int cal = (units*7);
                printf("%d Bill : ",cal);
            }
            else if(units>100){
                int cal = (100*7)+((units-100)*10);
                printf("%d Bill : ",cal);
            }
    }
    return 0;
}
