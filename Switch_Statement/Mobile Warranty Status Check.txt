#include <stdio.h>

int main() {
    int warrantyCategory;


    scanf("%d", &warrantyCategory);

    switch (warrantyCategory) {
        case 1:  
            printf("Under Warranty");
            break;

        case 2:  
            printf("Limited Warranty");
            break;

        case 3:  
            printf("Out of Warranty");
            break;

        default:
            printf("Invalid warranty category");
    }

    return 0;
}
