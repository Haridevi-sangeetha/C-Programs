#include<stdio.h>

void prime(){
    int num;
    scanf("%d",&num);
    int count=0;
    for(int i=1;i<=num;i++){
        if(num%i==0){
            count++;
        }
    }
    if(count==2){
        printf("Prime");
    }
    else{
        printf("Not Prime");
    }
}

int main(){
    prime();
    return 0;
}
