#include<stdio.h>
void palindrome(){
    int num;
    scanf("%d",&num);
    int temp=num;
    int rev=0;
    while(temp!=0){
        int id=temp%10;
        rev=(rev*10)+id;
        temp=temp/10;
    }
    if(rev==num){
        printf("Palindrome");
    }
    else{
        printf("Not Palindrome");
    }
}

int main(){
    palindrome();
    return 0;
}
