
#include<stdio.h>

int main(){
    int size;
    scanf("%d",&size);
    int arr[size];
    int arr1[size];
    int count=0;
    for(int i=0;i<size;i++){
        scanf("%d",&arr[i]);
    }
    int largest = 0;
    for(int i=(size-1);i>=0;i--){
        if(arr[i]>largest){
            largest=arr[i];
            arr1[count++]=largest;
        }
    }
    for(int i=(count-1);i>=0;i--){
        printf("%d ",arr1[i]);
    }
}
