#include<stdio.h>

int main(){
    int size;
    scanf("%d",&size);
    int arr[size];
    int arr1[size];
    for(int i=0;i<size;i++){
        scanf("%d",&arr[i]);
    }
    int count;
    for(int i=0;i<size;i++){
        count=0;
        for(int j=0;j<size;j++){
            if(arr[i]==arr[j]){
                count+=1;
            }
        }
        if(count>1){
            arr1[i]=arr[i]*arr[i];
        }
        else{
            arr1[i]=arr[i];
        }
    }
    for(int i=0;i<size;i++){
        printf("%d ",arr1[i]);
    }
    return 0;
}
