#include<stdio.h>

int main()
{
    int n;
    scanf("%d", &n);

    int arr[n];
    for(int i = 0; i < n; i++)
    {
        scanf("%d", &arr[i]);
    }

    int currentStreak = 1;
    int maxStreak = 1;

    for(int i = 1; i < n; i++)
    {
        if(arr[i] > arr[i-1])
        {
            currentStreak++;
        }
        else
        {
            currentStreak = 1;
        }

        if(currentStreak > maxStreak)
        {
            maxStreak = currentStreak;
        }
    }

    printf("%d", maxStreak);
    return 0;
}
