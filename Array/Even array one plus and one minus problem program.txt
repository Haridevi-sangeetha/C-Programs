
#include<stdio.h>

int main(){
    int size;
    scanf("%d",&size);
    int arr[size];
    int arr1[size];
    for(int i=0;i<size;i++){
        scanf("%d",&arr[i]);
    }
    int k=0,j=0;
    for(int i=0;i<size;i++){
        if(arr[i]>0){
            arr1[k]=arr[i];
            k++;
        }
        else if(arr[i]<0){
            arr1[size-1-j]=arr[i];
            j++;
        }
    }
    
    
    for(int i=0;i<size;i++){
        printf("%d ",arr1[i]);
    }
    printf("\n");
    int count=0;
    int left=0;
    int right=size-1;
    for(int i=0;i<size-1;i++){
        printf("%d ",arr1[left]);
        printf("%d ",arr1[right]);
        left++;
        right--;
        count+=1;
        if(count==size/2)
              break;
    }
    return 0;
}
