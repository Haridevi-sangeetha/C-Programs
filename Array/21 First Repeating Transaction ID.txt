#include<stdio.h>

int main()
{
    int size;
    scanf("%d", &size);

    int arr[size];
    for(int i = 0; i < size; i++)
    {
        scanf("%d", &arr[i]);
    }

    int found = 0;
    int result = -1;

    for(int i = 0; i < size; i++)
    {
        for(int j = i + 1; j < size; j++)
        {
            if(arr[i] == arr[j])
            {
                result = arr[i];
                found = 1;
                break; 
            }
        }
        if(found == 1)
            break;  
    }

    if(found == 1)
        printf("%d", result);
    else
        printf("-1");

    return 0;
}
