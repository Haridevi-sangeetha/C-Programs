
#include<stdio.h>

int main(){
    int size;
    printf("Enter the size of the array : ");
    scanf("%d",&size);
    int k;
    printf("Enter the K Position want yo move : ");
    scanf("%d",&k);
    int arr[size];
    for(int i=0;i<size;i++){
        scanf("%d",&arr[i]);
    }
    // k=k%2;
    for(int i=0;i<k;i++){
        int first = arr[0];
        for(int j=0;j<size;j++){
            int temp = arr[j];
            arr[j]=arr[j+1];
            arr[j+1]=temp;
        }
        arr[size-1]=first;
    }
    for(int i=0;i<size;i++){
        printf("%d ",arr[i]);
    }
    return 0;
}
