#include<stdio.h>

int main(){
    int size,target;
    scanf("%d",&size);
    scanf("%d",&target);
    int arr[size];
    int count=0;
    for(int i=0;i<size;i++){
        scanf("%d",&arr[i]);
        if(target==arr[i]){
            count=count+1;
        }
    }
    printf("%d",count);
}
